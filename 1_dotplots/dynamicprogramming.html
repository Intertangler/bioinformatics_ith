    #print i,j,int(matrix_i[i][j]), int(matrix_j[i][j]),i_last,j_last
    try:
        if i == i_last:
            aligned_long += '-'
        else:
            aligned_long += longstring[i]
    except:
        aligned_long += '-'
    try:
        if j == j_last:
            aligned_short += '-'
        else:
            aligned_short += shortstring[j]
    except:
        aligned_short += '-'

    if i == 0 and j == 0:
        try:
            if i == i_last:

            aligned_long += '-'

            else:

            aligned_short += '-'
